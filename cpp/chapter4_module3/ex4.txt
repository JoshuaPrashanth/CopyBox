class Median {
public:
    // Two numbers
    double CalculateMedian(int a, int b) {
        return (double)(a + b) / 2;
    }

    // Three numbers
    double CalculateMedian(int a, int b, int c) {
        int arr[3] = {a, b, c};
        sort(arr, arr + 3);
        return arr[1]; // middle element
    }

    // Four numbers
    double CalculateMedian(int a, int b, int c, int d) {
        int arr[4] = {a, b, c, d};
        sort(arr, arr + 4);
        return (double)(arr[1] + arr[2]) / 2; // average of middle two
    }

    // Five numbers
    double CalculateMedian(int a, int b, int c, int d, int e) {
        int arr[5] = {a, b, c, d, e};
        sort(arr, arr + 5);
        return arr[2]; // middle element
    }
};
