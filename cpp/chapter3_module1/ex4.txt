bool IsPalindrome(string str) {
    string reversed = "";  // empty string to hold reversed version

    // populate reversed string
    for (int i = str.length() - 1; i >= 0; i--) {
        reversed += str[i];
    }

    // compare original and reversed strings
    return str == reversed;
}
