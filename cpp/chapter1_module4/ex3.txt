#include <iostream>
using namespace std;
int main(int argc, char** argv) {
  int a = stoi(argv[1]);
  int b = stoi(argv[2]);
  if (a > b) {
    int c = b;
    b = a;
    a = c;
  }
  
  int sum = 0;

  for (int i = a; i <= b; i++) {
    sum += i;
  }
  cout << sum << endl; // only print the sum
  return 0;
}
