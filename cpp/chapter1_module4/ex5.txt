#include <iostream>
using namespace std;

int main() {
    for (int i = 1; i <= 5; i++) {
        // Print leading dots
        for (int j = 5 - i; j > 0; j--) {
            cout << ".";
        }

        // Print the number i
        cout << i << "\n";
    }

    return 0;  // Proper return statement
}
