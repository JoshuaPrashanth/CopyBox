#include <iostream>
#include <vector>
using namespace std;

//add function definitions below this line

int ListSum(const vector<int>& nums) {
    // Base case: if the vector is empty, sum is 0
    if (nums.empty()) {
        return 0;
    }
    // Recursive case: sum first element + sum of the rest
    vector<int> rest(nums.begin() + 1, nums.end());
    return nums[0] + ListSum(rest);
}

//add function definitions above this line

int main(int argc, char** argv) {
    vector<int> nums;
    for (int i = 1; i < argc; i++) {
        nums.push_back(stoi(argv[i]));
    }
    cout << ListSum(nums) << endl;
    return 0;
}
